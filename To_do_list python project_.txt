# Simple To-Do List in Python

tasks = []

def show_menu():
    print("\nðŸ“Œ To-Do List Menu:")
    print("1. Add Task")
    print("2. View Tasks")
    print("3. Remove Task")
    print("4. Exit")

while True:
    show_menu()
    choice = input("Enter your choice (1-4): ")

    if choice == "1":
        task = input("Enter new task: ")
        tasks.append(task)
        print(f"âœ… '{task}' added to list.")

    elif choice == "2":
        print("\nðŸ“‹ Your Tasks:")
        if not tasks:
            print("No tasks yet!")
        else:
            for i, task in enumerate(tasks, start=1):
                print(f"{i}. {task}")

    elif choice == "3":
        print("\nðŸ“‹ Your Tasks:")
        for i, task in enumerate(tasks, start=1):
            print(f"{i}. {task}")
        task_no = int(input("Enter task number to remove: "))
        if 0 < task_no <= len(tasks):
            removed = tasks.pop(task_no - 1)
            print(f"âŒ '{removed}' removed from list.")
        else:
            print("Invalid task number!")

    elif choice == "4":
        print("ðŸ‘‹ Exiting To-Do List. Bye!")
        break

    else:
        print("âš  Invalid choice, try again.")
